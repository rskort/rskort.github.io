Flatten Nested Lists in Python

A simple and efficient way to flatten nested lists using list comprehension. This example shows how to handle lists of arbitrary depth. 

def flatten(nested_list):
    """Flatten a nested list of arbitrary depth."""
    result = []
    for item in nested_list:
        if isinstance(item, list):
            result.extend(flatten(item))
        else:
            result.append(item)
    return result

# Example usage
nested = [[1, 2], [3, [4, 5]], 6]
flat = flatten(nested)
print(flat)  # Output: [1, 2, 3, 4, 5, 6]

Function Memoization in JavaScript

A decorator pattern for caching function results, useful for expensive computations. This prevents redundant calculations by storing previously computed values. 

function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      return cache.get(key);
    }
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Example: memoized Fibonacci
const fib = memoize(function(n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
});

console.log(fib(30));  // Fast computation
Language: JavaScript

Safe Dictionary Access with Defaults

A utility class for safely accessing nested dictionary values without raising KeyError exceptions. Useful when working with unpredictable or partially structured data. 

class SafeDict:
    """Access nested dictionary values safely."""
    def __init__(self, data):
        self.data = data
    
    def get_path(self, *keys, default=None):
        """Get value from nested dict using dot notation."""
        current = self.data
        for key in keys:
            if isinstance(current, dict):
                current = current.get(key)
            else:
                return default
            if current is None:
                return default
        return current

# Example usage
data = {'user': {'profile': {'name': 'Alice'}}}
safe = SafeDict(data)
name = safe.get_path('user', 'profile', 'name')  # 'Alice'
email = safe.get_path('user', 'contact', 'email', default='N/A')  # 'N/A'

Binary Search Implementation

A classic binary search algorithm with $O(\log n)$ time complexity. Requires a sorted array and efficiently finds target values. 

def binary_search(arr, target):
    """
    Binary search for target in sorted array.
    Returns index if found, -1 otherwise.
    """
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1

# Example
arr = [1, 3, 5, 7, 9, 11, 13]
idx = binary_search(arr, 7)  # Returns 3

Python String Manipulation Utilities

Common string operations including slug generation and case conversion. These utilities are helpful for text processing and URL generation. 

import re

def slugify(text):
    """Convert text to a URL-friendly slug."""
    text = text.lower().strip()
    text = re.sub(r'[^\w\s-]', '', text)
    text = re.sub(r'[-\s]+', '-', text)
    return text.strip('-')

def camel_to_snake(name):
    """Convert camelCase to snake_case."""
    s1 = re.sub('(.)([A-Z][a-z]+)', r'\1_\2', name)
    return re.sub('([a-z0-9])([A-Z])', r'\1_\2', s1).lower()

# Examples
print(slugify("Hello World!"))  # hello-world
print(camel_to_snake("myFunctionName"))  # my_function_name
